\section{Motivation}
% ═════════════════════════════════════════════════════════════════════════════

% ── Slide 3 — The Gap (text) ─────────────────────────────────────────────────
\begin{frame}{The Security Gap in Smart Grid Hosts}
  Smart grids converge IP networks with Windows-based SCADA and RTU hosts,
  creating an IT/OT boundary actively exploited by adversaries --- Stuxnet~(2010),
  Ukrainian grid attacks~(2015, 2016), Colonial Pipeline~(2021).

  \vspace{0.7em}
  \begin{itemize}
    \item NERC CIP-007 \emph{mandates} event logging on all applicable cyber assets
    \item IEC~62351 specifies audit trails for power system communications
    \pause
    \item \textbf{These logs are collected for compliance but never used for automated
          intrusion detection}
  \end{itemize}

  \vspace{0.7em}
  \pause
  \textbf{Goal:} apply machine learning to Windows Event Logs at Purdue
  Levels~1--3 to detect malicious host sessions in real time.
  \note{Core motivation: regulatory logging is mandated, terabytes of Windows Event
        Log data exist on SCADA hosts, but zero automated detection is performed on them.}
\end{frame}

% ── Slide 5 — Related Work ───────────────────────────────────────────────────
\begin{frame}{Prior Work and Research Gap}
  Three research streams inform this work --- each leaves a gap we fill:

  \vspace{0.5em}
  \textbf{Log anomaly detection} (Drain, DeepLog, LogAnomaly) targets distributed
  IT logs --- HDFS, syslog, OpenStack --- not Windows Security Event IDs in OT.

  \vspace{0.5em}
  \textbf{ICS/SCADA IDS} (Carcano, Goldenberg, SWaT dataset) monitors network
  traffic and physical process measurements, not host-level audit logs.

  \vspace{0.5em}
  \textbf{Windows malware detection} (Achmad 2025, \"{O}nal 2025, Zhen 2025) focuses
  on IT endpoints, not OT hosts with ICS-specific event semantics.

  \vspace{0.6em}
  \begin{block}{Confirmed gap \cite{mitchell2014survey}}
    Host-level IDS is rare in the cyber-physical systems literature.
    No prior work targets Windows Event Logs at Purdue L1--L3 in smart grids.
    \textbf{This work is the first such framework.}
  \end{block}
  \note{Three streams, three gaps. Mitchell and Chen 2014 directly confirms the absence
        of host-level IDS in CPS. Emphasise the OT/IT semantic mismatch.}
\end{frame}

\section{Threat Model}
% ═════════════════════════════════════════════════════════════════════════════

% ── Slide 6 — Threat Model: adversary ────────────────────────────────────────
\begin{frame}{Threat Model: Adversary and Goals}
  \begin{block}{Starting Position}
    Foothold on Level~3 IT host via phishing or supply-chain compromise.
    Valid or cracked domain credentials.
    Network access across a misconfigured IT/OT DMZ.
  \end{block}

  \vspace{0.8em}
  \begin{block}{Attack Chain}
    \centering
    \large
    Persist $\;\longrightarrow\;$ Escalate privileges $\;\longrightarrow\;$
    Execute disruptive payload

    \vspace{0.3em}
    \normalsize
    (rogue PLC commands \textit{or} denial of operator visibility)
  \end{block}
  \note{Adversary starts with a Level 3 foothold and valid credentials. The goal is
        to reach the OT layer and cause physical impact.}
\end{frame}

% ── Slide 7 — Threat Model: attack patterns ──────────────────────────────────
\begin{frame}{Threat Model: MITRE ATT\&CK ICS Patterns}
  Four attack categories modelled, each producing a distinct Windows Event Log signature:

  \vspace{0.6em}
  \begin{enumerate}
    \item \textbf{Lateral movement} --- 3--8 consecutive Event~4625 logon failures from an
          unexpected source IP, followed by a 4624 success

    \vspace{0.3em}
    \item \textbf{Persistence} --- Event~7045 (service install) on a SCADA host, followed
          by 7036 (state change) and 4688 (process creation)

    \vspace{0.3em}
    \item \textbf{Privilege escalation} --- Event~4672 (special privileges) immediately
          preceding 4688 with an anomalous binary (\texttt{mimikatz.exe},
          \texttt{psexec.exe})

    \vspace{0.3em}
    \item \textbf{Reconnaissance} --- burst of 8--20 Event~4663 (object access) records
          within 30~seconds on HMI or RTU hosts
  \end{enumerate}
  \note{Each pattern maps to MITRE ATT&CK for ICS. Evasive variants slow down lateral
        movement and reconnaissance to make single-feature detection harder.}
\end{frame}

\section{Data and Features}
% ═════════════════════════════════════════════════════════════════════════════

% ── Slide 8 — Synthetic Data: justification + stats ──────────────────────────
\begin{frame}{Synthetic Dataset: Justification and Statistics}
  No publicly available Windows Event Log dataset exists for smart grid SCADA or
  RTU environments~\cite{conti2021survey,nafees2023}. Synthetic generation is established
  practice: Goh et al.\ \cite{goh2016swat} built the SWaT testbed for the same reason;
  Chen et al.\ \cite{chen2022} used APT emulation when real logs were unavailable.

  \vspace{0.6em}
  \begin{itemize}
    \item \textbf{6{,}000 sessions:} 5{,}000 benign $+$ 1{,}000 malicious (5:1 ratio)
    \item \textbf{89{,}288 total events} across 13 Windows Event IDs
    \item \textbf{7 smart grid hostnames:} SCADA-HMI-01, RTU-CTRL-03,
          \texttt{eng\_station}, \texttt{historian}, \ldots
    \item \textbf{Evasive variants:} rate-limited lateral movement (60--300~s between
          failures) and slow reconnaissance (2--9~s burst intervals) prevent trivial
          single-feature separation
  \end{itemize}
  \note{Justify synthetic data up front. Evasive variants are the key design decision
        that makes the evaluation realistic rather than circular.}
\end{frame}

% ── Slide 10 — Feature Engineering ───────────────────────────────────────────
\begin{frame}{Session-Level Feature Engineering}
  Events grouped into sessions; each session produces a \textbf{24-dimensional vector}
  with no feature scaling required:

  \vspace{0.4em}
  \begin{table}
    \centering
    \begin{tabular}{llc}
      \toprule
      Group & Features & Dim. \\
      \midrule
      Event frequency counts  & count per event ID $+$ total                  & 14 \\
      Time-delta statistics   & mean, std, min, max inter-event gap (seconds) &  4 \\
      Host/user cardinality   & unique sources, users, hostnames per session   &  3 \\
      Rare-event flags        & binary: event in $<$5\% of benign sessions     & var \\
      Sequence entropy        & $H = -\sum_e p_e \log_2 p_e$                  &  1 \\
      \bottomrule
    \end{tabular}
  \end{table}

  \vspace{0.4em}
  \begin{block}{Design rationale}
    Mixed feature types handled natively by tree ensembles without normalisation.
    Session abstraction avoids deep sequence model latency while capturing temporal
    burst structure through time-delta statistics.
  \end{block}
  \note{Five groups, 24 dimensions. Entropy alone reaches F1 = 0.77 in isolation,
        confirming real signal. The session abstraction is what makes SIEM integration
        feasible at sub-second latency.}
\end{frame}

% ── Slide 11 — Classification Results ────────────────────────────────────────
\begin{frame}{Classification Results: Six-Method Comparison}
  Random Forest (class\_weight=balanced, RandomizedSearchCV, 5-fold stratified CV)
  versus five baselines:

  \vspace{0.3em}
  \begin{table}
    \centering
    \small
    \begin{tabular}{lcccc}
      \toprule
      Method & Precision & Recall & F1 (macro) & ROC-AUC \\
      \midrule
      Isolation Forest (unsupervised)   & 0.611 & 0.594 & 0.601 & 0.735 \\
      Logistic Regression               & 0.917 & 0.974 & 0.942 & 0.994 \\
      SVM (RBF kernel)                  & 0.818 & 0.881 & 0.844 & 0.951 \\
      Decision Tree                     & 0.982 & 0.987 & 0.984 & 0.988 \\
      XGBoost                           & 0.997 & 0.999 & 0.998 & 1.000 \\
      \textbf{Random Forest (proposed)} & \textbf{1.000} & \textbf{1.000} & \textbf{0.999} & \textbf{1.000} \\
      \bottomrule
    \end{tabular}
  \end{table}

  \vspace{0.3em}
  \begin{columns}[T]
    \begin{column}{0.52\textwidth}
      \small RF stability across 5 independent seeds:
      F1 $= 0.999 \pm 0.001$ $\;\cdot\;$ AUC $= 1.000 \pm 0.000$.\\
      All metrics computed from the actual pipeline output.
    \end{column}
    \begin{column}{0.44\textwidth}
      \begin{block}{\small No-label lower bound}
        \small Isolation Forest F1 = 0.601 shows detection difficulty
        without any labelled attack sessions.
      \end{block}
    \end{column}
  \end{columns}
  \note{Six baselines from unsupervised to boosted ensemble. Multi-seed std = 0.001
        rules out a lucky split. All numbers are real pipeline output.}
\end{frame}

% ── Slide 13 — Ablation ──────────────────────────────────────────────────────
\begin{frame}{Ablation Study: Feature Group Contribution}
  \begin{columns}[T]
    \begin{column}{0.46\textwidth}
      \begin{table}
        \centering
        \small
        \begin{tabular}{lcc}
          \toprule
          Removed Group & F1 & $\Delta$F1 \\
          \midrule
          None (full model)         & 1.000 & ---      \\
          Group~1: Event counts     & 0.944 & $-0.056$ \\
          Group~2: Time-delta stats & 0.972 & $-0.028$ \\
          Group~3: Cardinality      & 1.000 & 0.000    \\
          Group~4: Rare-event flags & 1.000 & 0.000    \\
          Group~5: Entropy          & 1.000 & 0.000    \\
          \bottomrule
        \end{tabular}
      \end{table}
    \end{column}
    \begin{column}{0.50\textwidth}
      Groups~3--5 show zero marginal delta because Groups~1 and~2 already fully
      separate the classes. Entropy alone achieves F1~$= 0.77$ in isolation ---
      real signal, but redundant once counts and timing are present.

      \vspace{0.5em}
      \begin{block}{Practical takeaway}
        If timestamp resolution is unavailable, preserve event counts first
        ($\Delta$F1 $= {-0.056}$ on removal). On noisier real-world logs,
        Groups~3--5 are expected to recover importance.
      \end{block}
    \end{column}
  \end{columns}
  \note{Real computed deltas. Groups 1 and 2 carry the discriminative load.
        Zero delta for Groups 3-5 means class-level redundancy, not uninformativeness.}
\end{frame}

\begin{frame}{Conclusion and Future Work}
  \begin{columns}[T]
    \begin{column}{0.54\textwidth}
      \textbf{Contributions:}
      \begin{itemize}
        \small
        \item First open-source framework for Windows Event Log malware
              detection in smart grid ICS (Purdue L1--L3)
        \item 24-feature session pipeline with evasive attack variants
        \item F1 $= 0.999 \pm 0.001$ (5 seeds); ROC-AUC $= 1.000$
        \item Event counts most discriminative ($\Delta$F1 $= {-0.056}$);
              timing statistics second ($-0.028$)

      \end{itemize}

      \begin{block}{Future Work}
        \begin{itemize}
          \small
          \item Real ICS logs from red-team exercises
          \item Multi-class ATT\&CK attribution
          \item Online learning for concept drift
          \item Adversarial robustness on real logs
        \end{itemize}
      \end{block}
    \end{column}
  \end{columns}
  \note{Five contributions, four future directions. Open-source reproducibility is
        the core community contribution given the absence of real ICS datasets.}
